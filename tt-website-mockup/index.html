Automated Bid Adjustment

bid_targets → meta_api.update_bid_cap()

Daily AI Creative Stack-Tests

# Tiger Tracks Growth OS — Daily 06:00 AM from tigertracks import shopify, meta_api, ltv # 1. Pull today's margin + cohort data cohort = shopify.get_new_customers("today") margins = shopify.get_margins("last_24h") # 2. Score cohort by predicted 90-day LTV scores = ltv.predict( cohort, features=["channel", "creative_hook", "aov", "device"] ) # 3. Segment by LTV tier high = scores[scores.ltv_p90 > 120] mid = scores[scores.ltv_p90.between(60, 120)] # 4. Auto-adjust bid caps in Meta API for adset in meta_api.active_adsets(): bid = calc_target_bid( ltv_score=scores.get(adset.id), target_cac=42.00, margin=margins.avg ) meta_api.set_bid_cap(adset.id, bid) print(f"✓ {len(adsets)} adsets updated · 06:02 AM")
